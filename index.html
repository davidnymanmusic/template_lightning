EJS

Embedded JavaScript



  <!DOCTYPE html>
  <html>
    <head>
      <title><%= title %></title>
      <link rel='stylesheet' href='/stylesheets/style.css' />
    </head>
    <body>
      <h1><%= title %></h1>
      <p>Welcome to <%= title %></p>
    </body>
  </html>

HBS

Handlebars



  <!DOCTYPE html>
  <html>
    <head>
      <title>{{title}}</title>
      <link rel='stylesheet' href='/stylesheets/style.css' />
    </head>
    <body>
      {{{body}}}
    </body>
  </html>
